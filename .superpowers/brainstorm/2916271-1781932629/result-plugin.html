完整数据流

# 使用方式
var source = FileAccess.get_file_as_string("res://player.gd")
var tokenizer = GDScriptTokenizer.new()
var tokens = tokenizer.tokenize(source)

var parser = GDScriptParser.new()
var ast = parser.parse(tokens)

var resolver = GDScriptSymbolResolver.new()
var result = resolver.resolve(ast, "res://player.gd")

# 查询
result.get_signal_flow("health_changed")
# → signal health_changed @line 3
#   ├─ connect: _on_health_changed @line 12 (hud.gd)
#   ├─ emit @line 8 (take_damage)
#   └─ emit @line 15 (heal)

result.get_callers_of("take_damage")
# → _on_body_entered @line 22, _process @line 30

result.get_variable_usages("hp")
# → def @line 2 (var hp := 100)
#   reads: line 6, 8, 14
#   writes: line 7, 15

EditorPlugin 集成

class_name GDScriptUtil extends EditorPlugin func _enter_tree(): # 注册到编辑器 add_tool_menu_item("GDScript Analysis", _on_analyze) # 可选: 添加编辑器底部面板 Dock add_control_to_bottom_panel(analysis_panel, "GDScript Analysis") func analyze_script(path: String) → GDScriptAnalysisResult: # 便捷入口 — 一键分析 var source = load(path).source_code var tokens = GDScriptTokenizer.new().tokenize(source) var ast = GDScriptParser.new().parse(tokens) return GDScriptSymbolResolver.new().resolve(ast, path)
