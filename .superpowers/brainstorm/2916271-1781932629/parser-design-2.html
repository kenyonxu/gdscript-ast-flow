parse() 主流程

func parse(tokens: Array) → ClassNode:
    self.tokens = tokens
    pos = 0

    root = ClassNode.new()

    # 处理文件级注解 (@tool, @icon...)
    while _peek().type == ANNOTATION:
        root.annotations.append(_parse_annotation())

    # 解析类体
    while _peek().type != EOF:
        root.members.append(_parse_class_member())

    return root

# 4.7: 独立 INDENT/DEDENT 令牌 — 与 Python 相同的模型
func _parse_suite() → SuiteNode:
    # 期望 COLON + NEWLINE + INDENT
    _expect(COLON)
    _expect(NEWLINE)
    if _peek().type == INDENT:
        _advance()
    suite = SuiteNode.new()
    while _peek().type not in [DEDENT, EOF]:
        suite.statements.append(_parse_statement())
        if _peek().type == NEWLINE:
            _advance()
    if _peek().type == DEDENT:
        _advance()        # 注意：分词器已经生成了 INDENT/DEDENT
    return suite
